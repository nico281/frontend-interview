  darkMode: 'class'
  Habilita dark mode basado en clase CSS (agrega/remueve dark en <html>).

  const [theme, setTheme] = useState<Theme>(() => {
    // 1. Lee localStorage
    // 2. Sino, detecta preferencia del SO
    // 3. Sino, default 'light'
  })

  useEffect(() => {
    if (theme === 'dark') {
      document.documentElement.classList.add('dark')
    } else {
      document.documentElement.classList.remove('dark')
    }
    localStorage.setItem('theme', theme)  // Persiste
  }, [theme])
